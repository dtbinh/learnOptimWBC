329  function create(varargin)
330  /* Creates the Doxygen documentation
331  *
332  * Parameters:
333  * varargin: Optional parameters for creation.
334  * open: Set to true if the documentation should be opened after
335  * successful compilation @type logical @default false
336  * latex: Set to true if `\text{\LaTeX}` output should be generated, too. @type logical
337  * @default false */
338 
339  /* % Preparations */
340 
341  ip.addParamValue('open',false,@islogical);
342  ip.addParamValue('latex',false,@islogical);
343  ip.parse(varargin[:]);
344  genlatex = ip.Results.latex;
345 
346  /* Check for correct setup */
347  cdir = MatlabDocMaker.getConfigDirectory;
348  srcdir = MatlabDocMaker.getSourceDirectory;
349  outdir = MatlabDocMaker.getOutputDirectory;
350  /* Check if doxygen config file template exists */
351  doxyfile_in = fullfile(cdir,MatlabDocMaker.DOXYFILE_TEMPLATE);
352  if exist(doxyfile_in," file ") ~= 2
353  error(" No doxygen configuration file template found at '%s' ",doxyfile_in);
354  end
355 
356  lstr = ;
357  if genlatex
358  lstr = " (+Latex) ";
359  end
360  fprintf([" Starting creation of doxygen/mtoc++ powered HTML%s documentation for '%s' (%s)\n "...
361  " Sources: %s\nOutput to: <a href='matlab:MatlabDocMaker.open'>%s</a>\nCreating config files... "],lstr,...
362  MatlabDocMaker.getProjectName,MatlabDocMaker.getProjectVersion,...
363  srcdir,outdir);
364 
365  /* Operation-system dependent strings */
366  strs = struct;
367  if isunix
368  strs.null= " /dev/null ";
369  strs.silencer= ;
370  elseif ispc
371  strs.null= " NUL ";
372  strs.silencer= " @ "; /* argh that took a while to remember.. */
373 
374  else
375  error(" Current platform not supported. ");
376  end
377 
378  /* Save current working dir and change into the KerMor home
379  * directory; only from there all classes and packages are
380  * detected properly. */
381  curdir = pwd;
382  cd(srcdir);
383 
384  /* Append the configuration file directory to the current PATH */
385  pathadd = [pathsep cdir];
386  setenv(" PATH ",[getenv(" PATH ") pathadd]);
387 
388  mtoc_conf = fullfile(cdir,MatlabDocMaker.MTOCPP_CONFIGFILE);
389  filter = sprintf(" %smtocpp ",strs.silencer);
390  if exist(mtoc_conf," file ")
391  if isunix
392  strs.filter= " mtocpp_filter.sh ";
393  strs.farg= " $1 ";
394  elseif ispc
395  strs.filter= " mtocpp_filter.bat ";
396  strs.farg= " %1 ";
397  else
398  error(" Current platform not supported. ");
399  end
400  /* % Creation part */
401  cdir = MatlabDocMaker.getConfigDirectory;
402  /* Create "configured" filter script for inclusion in doxygen */
403  filter = fullfile(cdir,strs.filter);
404  f = fopen(filter," w ");
405  fprintf(f," %smtocpp %s %s ",strs.silencer,strs.farg,mtoc_conf);
406  fclose(f);
407  if isunix
408  unix([" chmod +x " filter]);
409  end
410  end
411 
412  /* % Prepare placeholders in the Doxyfile template */
413  m = [" _OutputDir_ " strrep(outdir," \ "," \\ "); ...
414  " _SourceDir_ " strrep(MatlabDocMaker.getSourceDirectory," \ "," \\ ");...
415  " _ConfDir_ " strrep(cdir," \ "," \\ ");...
416  " _ProjectName_ " MatlabDocMaker.getProjectName; ...
417  " _ProjectDescription_ " MatlabDocMaker.getProjectDescription; ...
418  " _ProjectLogo_ " strrep(MatlabDocMaker.getProjectLogo," \ "," \\ "); ...
419  " _ProjectVersion_ " MatlabDocMaker.getProjectVersion; ...
420  " _MTOCFILTER_ " strrep(filter," \ "," \\ "); ...
421  ];
422 
423  /* Check for latex extra stuff */
424  texin = fullfile(cdir,MatlabDocMaker.LATEXEXTRAS_TEMPLATE);
425  latexextras = ;
426  if exist(texin," file ") == 2
427  latexstr = strrep(fileread(texin)," _ConfDir_ ",strrep(cdir," \ "," / "));
428  latexextras = fullfile(cdir," latexextras.sty ");
429  fid = fopen(latexextras," w+ "); fprintf(fid," %s ",latexstr); fclose(fid);
430  end
431  /* Always use "/" for latex usepackage commands, so replace "\" (effectively windows
432  * only) by "/", and dont pass the extension ".sty" as latex automatically adds it. */
433  m(end+1,:) = [" _LatexExtras_ " strrep(latexextras(1:end-4)," \ "," / ")];
434  L = " NO ";
435  if genlatex
436  L = " YES ";
437  end
438  m(end+1,:) = [" _GenLatex_ ",L];
439 
440  /* Check how to set the HAVE_DOT flag */
441  [s, ~] = system(" dot -V ");
442  if s == 0
443  HD = " YES ";
444  else
445  HD = " NO ";
446  fprintf(" no 'dot' found... ");
447  end
448  m(end+1,:) = [" _HaveDot_ ",HD];
449 
450  /* Read, replace & write doxygen config file */
451  doxyfile = fullfile(cdir," Doxyfile ");
452  doxyconfstr = regexprep(fileread(doxyfile_in),m(:,1),m(:,2));
453  fid = fopen(doxyfile," w "); fprintf(fid," %s ",doxyconfstr); fclose(fid);
454 
455  /* Fix for unix systems where the MatLab installation uses older
456  * GLIBSTD libraries than doxygen/mtoc++ */
457  ldpath = ;
458  if isunix
459  ldpath = " LD_LIBRARY_PATH= ";
460  end
461  /* Call doxygen */
462  fprintf(" running doxygen with mtoc++ filter... ");
463  [~,warn] = system(sprintf(" %sdoxygen '%s' 1>%s ",ldpath, doxyfile, strs.null));
464 
465  /* Postprocess */
466  fprintf(" running mtoc++ postprocessor... ");
467  [~,postwarn] = system(sprintf(" %smtocpp_post '%s' 1>%s ",ldpath,...
468  outdir, strs.null));
469  if ~isempty(postwarn)
470  warn = [warn sprintf(" mtoc++ postprocessor messages:\n ") postwarn];
471  end
472 
473  /* Create latex document if desired */
474  if genlatex
475  oldd = pwd;
476  latexerr = false;
477  latexdir = fullfile(outdir," latex ");
478  if exist(latexdir," dir ") == 7
479  if exist(fullfile(latexdir," refman.tex ")," file ") == 2
480  fprintf(" compiling LaTeX output... ");
481  cd(latexdir);
482  [s, latexmsg] = system(" make ");
483  if s ~= 0
484  warn = [warn sprintf(" LaTeX compiler output:\n ") latexmsg];
485  latexerr = true;
486  end
487  else
488  fprintf(" cannot compile LaTeX output: no refman.tex found... ");
489  end
490  end
491  cd(oldd);
492  end
493 
494  /* Tidy up */
495  fprintf(" cleaning up... ");
496  if isfield(strs," filter ")
497  delete(filter);
498  end
499  if ~isempty(latexextras)
500  delete(latexextras);
501  end
502  delete(doxyfile);
503 
504  /* % Post generation phase */
505  cd(curdir);
506  /* Restore PATH to previous value */
507  curpath = getenv(" PATH ");
508  setenv(" PATH ",curpath(1:end-length(pathadd)));
509  fprintf(" done!\n ");
510 
511  /* Process warnings */
512  showchars = 800;
513  warn = strtrim(warn);
514  if ~isempty(warn)
515  dispwarn = [warn(1:min(showchars,length(warn))) " [...] "];
516  fprintf(" First %d characters of warnings generated during documentation creation:\n%s\n ",showchars,dispwarn);
517  /* Write to log file */
518  log = fullfile(outdir," warnings.log ");
519  f = fopen(log," w "); fprintf(f," %s ",warn); fclose(f);
520  fprintf(2," MatlabDocMaker finished with warnings!\n ");
521  fprintf(" Complete log file at <a href='matlab:edit( "" %s "" )'>%s</a>.\n ",log,log);
522  /* Check for latex log file */
523  log = fullfile(outdir," _formulas.log ");
524  if exist(log," file ")
525  fprintf(" Found LaTeX formula log file. Check <a href='matlab:edit( "" %s "" )'>%s</a> for any errors.\n ",log,log);
526  end
527  /* Check for errors on latex generation */
528  if genlatex && latexerr
529  log = fullfile(latexdir," refman.log ");
530  fprintf(" There have been errors with LaTeX compilation. See log file at <a href='matlab:edit( "" %s "" )'>%s</a>.\n ",log,log);
531  end
532  else
533  fprintf(" MatlabDocMaker finished with no warnings!\n ");
534  end
535 
536  /* Open index.html if wanted */
537  if ip.Results.open
538  MatlabDocMaker.open;
539  end
540  end
