330 function [fout, vout, cout] = rndread(filename)
331  /* Reads CAD STL ASCII files, which most CAD programs can export.
332  * Used to create Matlab patches of CAD 3D data.
333  * Returns a vertex list and face list, for Matlab patch command.
334  *
335  * filename = 'hook.stl'; % Example file.
336  * */
337  fid=fopen(filename, " r "); /* Open the file, assumes STL ASCII format. */
338 
339  if fid == -1
340  error(" File could not be opened, check name or path. ")
341  end
342  /*
343  % Render files take the form:
344  *
345  *solid BLOCK
346  * color 1.000 1.000 1.000
347  * facet
348  * normal 0.000000e+00 0.000000e+00 -1.000000e+00
349  * normal 0.000000e+00 0.000000e+00 -1.000000e+00
350  * normal 0.000000e+00 0.000000e+00 -1.000000e+00
351  * outer loop
352  * vertex 5.000000e-01 -5.000000e-01 -5.000000e-01
353  * vertex -5.000000e-01 -5.000000e-01 -5.000000e-01
354  * vertex -5.000000e-01 5.000000e-01 -5.000000e-01
355  * endloop
356  * endfacet
357  *
358  * The first line is object name, then comes multiple facet and vertex lines.
359  * A color specifier is next, followed by those faces of that color, until
360  * next color line.
361  * */
362  CAD_object_name = sscanf(fgetl(fid), " %*s %s "); /* CAD object name, if needed.
363  * %Some STLs have it, some don't. */
364 
365  vnum=0; /* Vertex number counter. */
366 
367  report_num=0; /* Report the status as we go. */
368 
369  VColor = 0;
370  /*
371  while feof(fid) == 0 % test for end of file, if not then do stuff */
372  tline = fgetl(fid); /* reads a line of data from file. */
373 
374  fword = sscanf(tline, " %s "); /* make the line a character string
375  * Check for color */
376 
377  if strncmpi(fword, " c ",1) == 1; /* Checking if a "C"olor line, as "C" is 1st char. */
378 
379  VColor = sscanf(tline, " %*s %f %f %f "); /* & if a C, get the RGB color data of the face. */
380 
381  end /* Keep this color, until the next color is used. */
382 
383  if strncmpi(fword, " v ",1) == 1; /* Checking if a "V"ertex line, as "V" is 1st char. */
384 
385  vnum = vnum + 1; /* If a V we count the # of V's */
386 
387  report_num = report_num + 1; /* Report a counter, so long files show status */
388 
389  if report_num > 249;
390  /* disp(sprintf('Reading vertix num: %d.',vnum)); */
391  report_num = 0;
392  end
393  v(:,vnum) = sscanf(tline, " %*s %f %f %f "); /* & if a V, get the XYZ data of it. */
394 
395  c(:,vnum) = VColor; /* A color for each vertex, which will color the faces. */
396 
397  end /* we "*s" skip the name "color" and get the data. */
398 
399  end
400  /* Build face list; The vertices are in order, so just number them.
401  * */
402  fnum = vnum/3; /* Number of faces, vnum is number of vertices. STL is triangles. */
403 
404  flist = 1:vnum; /* Face list of vertices, all in order. */
405 
406  F = reshape(flist, 3,fnum); /* Make a "3 by fnum" matrix of face list data.
407  *
408  * Return the faces and vertexs.
409  * */
410 
411  fout = F^t; /* Orients the array for direct use in patch. */
412 
413  vout = v^t; /* " */
414 
415  cout = c^t;
416  /*
417  fclose(fid); */
418 end

421 function create_tiled_floor(opt)
422 
423  xmin = opt.ws(1);
424  xmax = opt.ws(2);
425  ymin = opt.ws(3);
426  ymax = opt.ws(4);
427 
428  /* create a colored tiled floor */
429  xt = xmin:opt.tilesize:xmax;
430  yt = ymin:opt.tilesize:ymax;
431  Z = opt.floorlevel*ones( numel(yt), numel(xt));
432  C = zeros(size(Z));
433  [r,c] = ind2sub(size(C), 1:numel(C));
434  C = bitand(r+c,1);
435  C = reshape(C, size(Z));
436  C = cat(3, opt.tile1color(1)*C+opt.tile2color(1)*(1-C), ...
437  opt.tile1color(2)*C+opt.tile2color(2)*(1-C), ...
438  opt.tile1color(3)*C+opt.tile2color(3)*(1-C));
439  [X,Y] = meshgrid(xt, yt);
440  surface(X, Y, Z, C, ...
441  " FaceColor "," texturemap ",...
442  " EdgeColor "," none ",...
443  " CDataMapping "," direct ");
444 end
